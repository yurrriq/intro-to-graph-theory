\begin{align*}
  1 + 2 + ... + (v-1) &= (1/2)v(v-1) \\
  &= E(K_{v}) & \text{(T2)} \\
  &= (v-1) + (v-2) + ... + (v - (v-1)) \\
  &= 1 + 2 + ... (v-1)
\end{align*}
$\hfill\square$

\marginnote[-7em]{%
  Imagine drawing $K_{v}$ by joining vertex $1$ to vertices $2$ through $v$,
  creating $v-1$ edges; then joining vertex $2$ to vertices $3$ through $v$,
  creating $v-2$ edges; and so on, i.e.
  $(v - 1) + (v - 2) + ... + (v - (v - 1))$, or equivalently,
  $1 + 2 + ... + (v - 1)$.%
}
