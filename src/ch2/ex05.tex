\begin{align*}
  V(P_{v})        &= \Set{ 1, 2, ..., v } \\
  E(P_{v})        &= \Set{ \Set{n-1,n} \colon n \in \Set{ 2, ..., v } } \\
  \abs*{E(P_{v})} &= v - 1 \\
  e               &= v - 1
\end{align*}
$\hfill\square$

\marginnote[-6em]{%
  The number of edges in a \gls{path graph} $P_{v}$, where $v \ge 2$, is
  given by the formula $e = v - 1$.%
}
