\[
\begin{prooftree}
  \infer0{ S(b) \coloneqq \Set{ m \in V \colon m \not\in S(m) } }
  \infer0{ b \in V }
  \infer2{ b \in S(b) \Leftarrow\!\Rightarrow b \not\in S(b) }
\end{prooftree}
\equiv
\begin{prooftree}
  \infer0{ R = \Set{ x \not\in x } }
  \infer1{ R \in R \Leftarrow\!\Rightarrow R \not\in R }
\end{prooftree}
\]
